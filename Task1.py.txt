class car:
    def __init__(self,brand,model,year,username):
        self.brand=brand
        self.model=model
        self.year=year
        self.username=username 
        
    def display (self):
        print(f"{self.brand} {self.model} {self.year} {self.username}")
        
c=car("Honda","Civic","2013","Rebirth")
c.display()
